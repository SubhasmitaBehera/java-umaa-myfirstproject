public class ArrayTraverse {
        public static void main(String[] args) {
            String arr[] = {"subhasmita","sangits","ritu", "sunil", "bibhu","amit","sumit"};
            int i = 0;
            while (i < arr.length)
            {
                System.out.println(arr[i]);
                i++;
            }
        }
}
